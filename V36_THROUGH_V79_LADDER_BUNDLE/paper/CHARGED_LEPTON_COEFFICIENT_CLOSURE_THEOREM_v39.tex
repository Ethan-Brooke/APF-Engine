\label{thm:charged-lepton-coefficient-closure-v39}

\[

        =\operatorname{diag}\!\left({1\over3},3,1\right).}

\end{theorem}

\begin{proof}

\[
\alpha H_1+\beta H_2

\[
        -\alpha+\beta=-\log N_c,
        \qquad
        -2\beta=\log N_c,
        \qquad
        \alpha+\beta=0.

\[
        \beta=-{1\over2}\log N_c,
        \qquad
        \alpha={1\over2}\log N_c,
\]
and these automatically satisfy the first equation.  Since $H_1$ and $H_2$
span the full diagonal trace-free generation plane, the solution is unique.
Substituting $N_c=3$ gives the displayed APF coefficient closure.  No physical
charged-lepton mass is used in the derivation.
\end{proof}
