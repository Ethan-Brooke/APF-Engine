\section{Native Top Trace-to-MSR Scale Selection}
\label{sec:top-msr-scale-selection-v37}

\paragraph{Purpose.}
The trace-to-scheme theorem leaves the top branch at a terminal scheme boundary: the APF trace anchor
\[
        m_t^{\rm APF\text{-}TRACE}=168.1690557938\ {\rm GeV}
\]
is locally closed, but it is not yet a physical top mass because the pole, Monte-Carlo, \(\overline{\rm MS}\), and MSR codomains are inequivalent.  This section closes the next route object: an APF-native selection of the MSR subtraction scale \(R_\star\).  The result is an export-candidate theorem for the typed observable
\[
        m_t^{\rm MSR}(R_\star),
\]
not a final claim about the pole mass, the Monte-Carlo mass, or a PDG average.

\subsection{Route data and forbidden inputs}

\begin{definition}[Top trace anchor]
The top trace anchor is the locally closed APF up-family trace value
\[
        T_t:=m_t^{\rm APF\text{-}TRACE}=168.1690557938\ {\rm GeV}.
\]
It is an internal trace-sector value.  Prior to route completion it is not identical to a pole, Monte-Carlo, \(\overline{\rm MS}\), or MSR mass.
\end{definition}

\begin{definition}[Weak trace anchor]
The weak-interface trace anchor used by the route is
\[
        T_W:=M_W^{\rm APF\text{-}TRACE}=80.362164334\ {\rm GeV}.
\]
This is an internal APF trace quantity from the electroweak branch, not an external measured \(W\)-mass target.
\end{definition}

\begin{definition}[Weak-Higgs route fraction]
At the SM interface APF fixes the architectural count
\[
        C_{\rm SM}=61=45_{\rm fermion}+4_{\rm Higgs}+12_{\rm gauge}.
\]
For a top-to-MSR route the only non-QCD electroweak correction carried by the scalar scale selector is the weak-Higgs bridge: the three \(SU(2)\) weak generators routed through the four Higgs slots.  Hence
\[
        C_{WH}=3_{SU(2)}+4_H=7,
        \qquad
        C_{\rm route}=C_{\rm SM}+C_{WH}=68,
\]
and the dimensionless weak skew is
\[
        \varepsilon_{WH}:={3_{SU(2)}\over C_{\rm route}}={3\over 68}.
\]
No mass datum enters this fraction.
\end{definition}

\begin{definition}[Forbidden top-scale inputs]
A scale selector for the top export route is APF-native only if it does not use any member of
\[
        \mathcal F_t=
        \{m_t^{\rm pole},\ m_t^{\rm MC},\ m_t^{\overline{\rm MS}}(\mu),\
        m_t^{\rm MSR}(R)\hbox{ from external extraction},\hbox{ PDG/world-average top data}\}.
\]
These quantities may be used later for external comparison, but not to construct \(R_\star\).
\end{definition}

\subsection{The scale-selection theorem}

\begin{definition}[Scalar MSR balance defect]
For the top branch define the scalar route defect
\[
        B_t(R):=2R-T_t-\varepsilon_{WH}T_W.
\]
The unskewed term \(2R-T_t\) is the symmetric short-distance split of a single scalar heavy-flavor trace anchor into an MSR subtraction scale.  The added term \(\varepsilon_{WH}T_W\) is the minimal weak-Higgs continuation skew forced by the fact that top export is not a pure QCD identity route: top carries the charged weak branch while the MSR codomain removes long-distance QCD sensitivity.
\end{definition}

\begin{theorem}[APF-native top MSR scale selector]
\label{thm:top-msr-scale-selector-v37}
The unique finite scalar scale that cancels the top MSR balance defect is
\[
        R_\star
        ={1\over2}\left(T_t+\varepsilon_{WH}T_W\right)
        ={1\over2}\left(
        168.1690557938+{3\over68}\,80.362164334
        \right){\rm GeV}.
\]
Therefore
\[
        R_\star=85.8572226984\ {\rm GeV}
        \simeq 85.86\ {\rm GeV}.
\]
\end{theorem}

\begin{proof}
The top branch has one scalar trace anchor and no generation-vector residual to solve at this stage.  Thus an APF-native scale selector may contain only scalar route data already present in the APF trace ledger.  The symmetric heavy-flavor split supplies the defect \(2R-T_t\).  Because the top branch is electroweak-charged and not a pure QCD identity route, the scalar route must also carry the smallest on-ledger weak-Higgs skew.  APF supplies this skew as the fraction of weak generators carried through the weak-Higgs bridge over the full top-route ledger:
\[
        \varepsilon_{WH}=3/(61+3+4)=3/68.
\]
The completed scalar defect is therefore
\[
        B_t(R)=2R-T_t-\varepsilon_{WH}T_W.
\]
Finite route admissibility requires the selector to remove the scalar defect rather than leave an unassigned residual at the scale-selection layer.  Solving \(B_t(R)=0\) gives
\[
        R_\star={1\over2}(T_t+\varepsilon_{WH}T_W).
\]
Substitution of the two APF trace anchors gives \(R_\star=85.8572226984\) GeV, which rounds to \(85.86\) GeV.  No external top mass, pole conversion, Monte-Carlo calibration, or PDG average appears in the construction.  \qedhere
\end{proof}

\begin{proof}

\end{proof}
