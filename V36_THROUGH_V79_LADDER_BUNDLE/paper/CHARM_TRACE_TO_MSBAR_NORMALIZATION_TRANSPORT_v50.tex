\begin{theorem}[Charm Schur-count normalization]

\[
K_{\rm SM}=61,\qquad K_c=16,\qquad K_b=3,\qquad N_{\rm gen}=3.
\]

\[

\[
\boxed{\kappa_c=\sqrt{\frac72}}.
\]
The APF charm export-candidate map is therefore
\[

=\sqrt{\frac72}\,m_c^{\rm TRACE}

\]
\end{theorem}
