
\section{Top MSR Multiloop Numeric Evaluator Implementation (v60)}

\paragraph{Purpose.}
Version v59 closed the full top-QCD conversion ledger. Version v60 implements the next object: a runnable numeric evaluator that computes exactly the loop orders for which explicit sourced coefficients are present and fails closed otherwise.

\begin{definition}[v60 source object]
The APF top source remains
\[
 m_t^{\rm MSR}(R_\star)=168.169055793800\,{\rm GeV},
 \qquad
 R_\star=85.857222698385\,{\rm GeV}.
\]
\end{definition}

\begin{definition}[Evaluator normalization]
The v60 evaluator uses the coefficient normalization
\[
 \Delta m_t^{(L)}(R_\star)
 =R_\star\sum_{n=1}^L c_n\left({\alpha_s(R_\star)\over \pi}\right)^n,
 \qquad
 m_t^{{\rm pole},L}=m_t^{\rm MSR}(R_\star)+\Delta m_t^{(L)}(R_\star).
\]
At leading order, the only closed coefficient is
\[
 c_1=C_F=1.333333333333.
\]
Higher coefficients \(c_2,c_3,c_4\) are not inferred from the top target; they must be imported from the declared MSR/QCD coefficient ledger.
\end{definition}

\begin{theorem}[LO executable closure]
With \(\alpha_s(M_Z)=0.1180\), \(M_Z=91.1876\,{\rm GeV}\), and \(n_f=5\), the v60 evaluator gives
\[
 \alpha_s(R_\star)=0.119032308653252,
\]
\[
 \Delta m_t^{\rm LO}=4.337410801564\,{\rm GeV},
\]
and therefore
\[
 \boxed{m_t^{\rm pole,LO}=172.506466595364\,{\rm GeV}.}
\]
This exactly reproduces the v58 witness as an executable route object.
\end{theorem}

\begin{theorem}[Fail-closed multiloop boundary]
If a user requests \(L\ge2\) using the v60 LO-only coefficient file, the evaluator must return
\[
 \texttt{TOP\_MSR\_MULTILOOP\_NUMERIC\_EVALUATOR\_FAIL\_CLOSED}
\]
with the missing coefficients named explicitly. % TOP_MSR_MULTILOOP_NUMERIC_EVALUATOR_FAIL_CLOSED
It may not infer \(c_2,c_3,c_4\) from a target top mass or from a residual minimization.
\end{theorem}

\begin{corollary}[No final-top overclaim]
Version v60 closes
\[
 \boxed{t:[P_{\rm LO\ numeric\ evaluator\ closed}+P_{\rm fail\text{-}closed\ multiloop\ boundary}].}
\]
It does not close
\[
 [P_{\rm physical\ final}],\quad [P_{\rm exact\ pole}],\quad [P_{\rm MC\ equality}],\quad [P_{\rm full\ multiloop\ numeric}].
\]
The next admissible operation is coefficient ingestion from the declared MSR/R-evolution/RunDec/on-shell--\(\overline{\rm MS}\) ledgers.
\end{corollary}
