\section{Top MSR-to-Pole One-Loop Transport Witness}
\label{sec:top-msr-to-pole-lo-transport-v58}

\paragraph{Purpose.}
Version v58 advances the top branch beyond the v57 codomain boundary by adding a
source-independent one-loop QCD transport witness from the banked APF
\(\mathrm{MSR}(R_\star)\) object to a pole/Monte-Carlo-proxy audit codomain.
This is a witness theorem, not a full top-mass finality theorem.

\begin{definition}[v58 external QCD witness ledger]
The APF source object is
\[
 m_t^{\rm MSR}(R_\star) \equiv m_t^{\rm APF\text{-}TRACE}
 =168.169055793800\,{\rm GeV},
 \qquad
 R_\star=85.857222698385\,{\rm GeV}.
\]
The external one-loop QCD ledger is
\[
 \alpha_s(M_Z)=0.1180\pm 0.0009,\quad
 M_Z=91.1876\,{\rm GeV},\quad
 n_f=5,\quad
 \beta_0=11-\frac{2n_f}{3}=7.666666666667,\quad
 C_F=\frac43.
\]
The audit-only external top comparator is the PDG direct-measurement average
\[
 m_t^{\rm direct,PDG2025}=172.56\pm 0.31\,{\rm GeV}.
\]
\end{definition}

\begin{definition}[One-loop MSR-to-pole witness map]
At one-loop order, with five-flavor running between \(M_Z\) and \(R_\star\), set
\[
 \alpha_s(R_\star)
 =\frac{\alpha_s(M_Z)}{1+\alpha_s(M_Z)\beta_0(2\pi)^{-1}
 \log(R_\star/M_Z)}
 =0.119032308653252.
\]
The v58 witness map is
\[
 \Phi_t^{\rm pole,LO}
 =m_t^{\rm MSR}(R_\star)+R_\star C_F\frac{\alpha_s(R_\star)}{\pi}.
\]
Numerically,
\[
 R_\star C_F\frac{\alpha_s(R_\star)}{\pi}
 =4.337410801564\,{\rm GeV},
\]
so
\[
 \boxed{\Phi_t^{\rm pole,LO}=172.506466595364\,{\rm GeV}.}
\]
\end{definition}

\begin{theorem}[v58 top one-loop transport admission]
The v58 top transport witness is source-independent and comparison-admissible at
one-loop order.  Its residual against the declared PDG 2025 direct audit value is
\[
 \Delta_t^{\rm LO}=-0.053533404636\,{\rm GeV},\qquad
 \frac{\Delta_t^{\rm LO}}{0.31}=-0.172688402053.
\]
With the propagated \(\alpha_s(M_Z)\) uncertainty
\[
 \sigma_{\alpha_s}=0.033371360178\,{\rm GeV}
\]
and the declared one-loop truncation scale
\[
 \sigma_{\rm trunc,LO}=0.219121139378\,{\rm GeV},
\]
the non-fit comparison envelope is
\[
 \varepsilon_{t,v58}
 =\sqrt{0.31^2+\sigma_{\alpha_s}^2+\sigma_{\rm trunc,LO}^2}
 =0.381087550836\,{\rm GeV},
\]
and
\[
 \frac{\Delta_t^{\rm LO}}{\varepsilon_{t,v58}}
 =-0.140475343576.
\]
Thus v58 upgrades the top route to
\[
 \boxed{t:[P_{\rm MSR}(R_\star)+P_{\rm pole,LO\ witness}]}
\]
while retaining the full multi-loop conversion boundary.
\end{theorem}

\begin{proof}[No-smuggling audit]
The PDG top average is not used to set \(R_\star\), \(\alpha_s(M_Z)\), \(\beta_0\),
\(C_F\), or the one-loop conversion formula.  The APF object \(m_t^{\rm MSR}(R_\star)\)
is fixed by the v37 scale-selector theorem.  The QCD constants and coefficients are
external ledger items.  The PDG direct value enters only after \(\Phi_t^{\rm pole,LO}\)
is computed.  Hence the route is an audit witness rather than an inverse fit.
\end{proof}

\paragraph{Nonclaims.}
Version v58 does not claim a physical-final top mass, exact pole mass, equality with
the Monte-Carlo mass, zero residual, or a full multi-loop \(\mathrm{MSR}\)-to-pole/
\(\overline{\rm MS}\) conversion.  Those require the next external ledger: loop
order \(L\ge2\), MSR coefficient convention, threshold/matching convention,
renormalon prescription, PDF/correlation handling, and full covariance.
